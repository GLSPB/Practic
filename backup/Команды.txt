--Команды 3 практической--
mkdir Lebedev

cd ~/Lebedev/

mkdir theory

cd theory

touch full.txt increment.txt diff.txt
echo Полное копирован — это процесс создания копии абсолютно всех данных оригина, независимо от времени и условий их изменений. > full.txt

echo Инкрементное копирование или добавочный бэкап — это метод копирования, при котором к исходной копии набора данных шаг за шагом приписываются дополнения, отражающие изменения в данных. > increment.txt

echo Дифференциальный бэкап - это тип резервного копирования файлов, при котором копируются не все исходные файлы, а только новые и измененные с момента создания предыдущей полной копии. > diff.txt

cd ~/Lebedev/

mkdir backup

cd backup

tar -cvf backup.tar ~/Lebedev/theory/

echo "Для создания инкрементных архивов tar использует специальный файл метаданных, куда записывает состояние архивируемых файлов, впоследствии набор данных сравнивается с метаданными и в архив помещаются только новые или измененные файлы, а метаданные обновляются." >> ~/Lebedev/theory/increment.txt

tar --create --gzip --file=incrembackup.tgz  --listed-incremental=incrembackup.snar ~/Lebedev/theory
